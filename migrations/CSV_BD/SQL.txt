CREATE TABLE public.autoevaluaciones (
  id_autoevaluacion integer GENERATED ALWAYS AS IDENTITY NOT NULL,
  fecha_inicio timestamp with time zone NOT NULL DEFAULT now(),
  fecha_fin timestamp with time zone,
  estado USER-DEFINED NOT NULL DEFAULT 'PENDIENTE'::estado_autoevaluacion,
  id_bodega integer NOT NULL,
  id_segmento integer,
  puntaje_final integer,
  id_nivel_sostenibilidad integer,
  estado_evidencia USER-DEFINED,
  CONSTRAINT autoevaluaciones_pkey PRIMARY KEY (id_autoevaluacion),
  CONSTRAINT autoevaluaciones_bodega_fk FOREIGN KEY (id_bodega) REFERENCES public.bodegas(id_bodega),
  CONSTRAINT autoevaluaciones_segmento_fk FOREIGN KEY (id_segmento) REFERENCES public.segmentos(id_segmento),

  id_localidad integer NOT NULL,
  telefono text NOT NULL CHECK (telefono ~ '^[0-9]+$'::text),
  email_institucional text NOT NULL CHECK (email_institucional ~~ '%@%'::text),
  fecha_registro timestamp with time zone NOT NULL DEFAULT now(),
